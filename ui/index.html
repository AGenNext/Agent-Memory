Vector Search

Semantic similarity search with HNSW indexes

Results
No results yet. Run a search to see results.
SurrealQL 
-- Create HNSW index
DEFINE INDEX doc_vec ON doc 
  FIELDS embedding HNSW 
  DIMENSION 384 DISTANCE COSINE;

-- Vector search
SELECT *, vector::distance::knn() AS score
FROM doc 
WHERE embedding <|5|> $embedding
ORDER BY score ASC;

SurrealQL 
-- Create entity
CREATE entity SET name='SurrealDB', type='company';

-- Create relation with properties
RELATE entity:alice -> knows -> entity:bob SET 
  since = d'2024-01-15',
  confidence = 0.95;

-- Traverse
SELECT <-related->entity.name AS connections 
FROM entity:alice;

Quick reference guide

Quick Start
# Connect
await db.connect()
await db.use({namespace: "test", database: "test"})
await db.signin({username: "root", password: "root"})

# CRUD
await db.create("user", {name: "Alice"})
await db.select("user")
await db.query("SELECT * FROM user")
Schema
DEFINE TABLE user SCHEMAFULL;
DEFINE FIELD name ON user TYPE string;
DEFINE FIELD email ON user TYPE string;
DEFINE FIELD age ON user TYPE int;

DEFINE INDEX user_email 
  ON user FIELDS email UNIQUE;
Relations
-- RELATE
RELATE user:alice -> follows -> user:bob
  SET since = d'2024-01-01'

-- Query
SELECT <-follows->user.name 
FROM user:alice
